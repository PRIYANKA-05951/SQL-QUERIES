A.SELECT
SELECT * FROM sales;

WHERE CLAUSE:
SELECT * 
FROM sales
WHERE CUSTOMER = 'ABC Stores';


ORDER BY:
SELECT CUSTOMER, GROSS_AMT
FROM sales
ORDER BY GROSS_AMT DESC;

GROUPBY:
SELECT CUSTOMER, SUM(GROSS_AMT) AS Total_Sales
FROM sales
GROUP BY CUSTOMER;


JOINS:
CREATE TABLE customers (
    CUSTOMER TEXT PRIMARY KEY
);

INSERT INTO customers
SELECT DISTINCT CUSTOMER FROM sales;

INNERJOIN:
SELECT c.CUSTOMER, s.Style, s.GROSS_AMT
FROM customers c
INNER JOIN sales s
ON c.CUSTOMER = s.CUSTOMER;

SUBQUERY:
SELECT CUSTOMER, GROSS_AMT
FROM sales
WHERE GROSS_AMT >
(SELECT AVG(GROSS_AMT) FROM sales);


AGGREGATE FUNCTIONS:
SELECT 
SUM(GROSS_AMT) AS Total_Revenue,
AVG(GROSS_AMT) AS Average_Sale
FROM sales;


CREATE VIEW:
CREATE VIEW monthly_sales AS
SELECT Months, SUM(GROSS_AMT) AS Monthly_Total
FROM sales
GROUP BY Months;

SELECT * FROM monthly_sales;

INDEX:
CREATE INDEX idx_customer
ON sales(CUSTOMER);
